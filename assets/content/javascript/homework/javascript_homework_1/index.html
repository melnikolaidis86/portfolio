A simple script that a prompt is asking for your first name and last name, displaying a welcome message using them as variables.

That's the code that was used to append the text entered in prompt to the above h3 tag.

            <script type="text/javascript">
            //Fetching the variables from prompt and HTML
            var name = prompt("Please enter your name: ");
            var lastName = prompt("Please enter your last name: ");
            var welcomeMessage = document.getElementById("welcomeMessage");
                
            //Creating text node
            var text = document.createTextNode(
            "Welcome to this website " + name + " " + lastName + "!!");
                
            //Appending text node into HTML
            welcomeMessage.appendChild(text);
            </script>
